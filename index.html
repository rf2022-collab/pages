import { createServer } from 'http'; import { spawn } from 'child_process'; // Windows 대응 환경 설정 및 셸 옵션 적용 const mcp = spawn('npx.cmd', ['-y', '@modelcontextprotocol/server-filesystem', 'D:\\AI\\RAG'], { shell: true }); mcp.stdout.setEncoding('utf8'); // 디버깅용 에러 출력 mcp.stderr.on('data', (data) => { console.error(`[MCP Error]: ${data}`); }); // 들어오는 응답들을 순서대로 처리하기 위한 큐(Queue) 관리 const pendingResponses = []; mcp.stdout.on('data', (data) => { // MCP 서버에서 데이터가 출력되면 대기 중인 HTTP 응답에 순서대로 매칭해 반환 const res = pendingResponses.shift(); if (res) { res.writeHead(200, { 'Access-Control-Allow-Origin': '*', 'Access-Control-Allow-Methods': 'GET, POST, OPTIONS', 'Access-Control-Allow-Headers': 'Content-Type, Authorization, Accept, mcp-protocol-version', 'Content-Type': 'application/json' }); res.end(data); } }); createServer((req, res) => { // 기본 CORS 헤더 대응 res.setHeader('Access-Control-Allow-Origin', '*'); res.setHeader('Access-Control-Allow-Methods', 'GET, POST, OPTIONS'); res.setHeader('Access-Control-Allow-Headers', 'Content-Type, Authorization, Accept, mcp-protocol-version'); res.setHeader('Content-Type', 'application/json'); if (req.method === 'OPTIONS') { res.writeHead(200); res.end(); return; } if (req.method === 'POST') { let body = ''; req.on('data', chunk => { body += chunk; }); req.on('end', () => { // 알림(Notification) 계열의 요청은 보낸 후 응답을 기다리지 않는 특성이 있음 if (body.includes('"notifications/initialized"')) { mcp.stdin.write(body + '\n'); res.writeHead(200); res.end(JSON.stringify({})); // 빈 응답 즉시 반환하여 세션 유지 return; } // 일반 요청들은 큐에 집어넣고 MCP 처리 결과를 대기 pendingResponses.push(res); mcp.stdin.write(body + '\n'); }); } else { res.writeHead(405); res.end(); } }).listen(9191, () => { console.log("🚀 연속 통신 규격이 완벽 적용된 로컬 MCP 브릿지가 9191 포트에서 대기 중입니다!"); });
